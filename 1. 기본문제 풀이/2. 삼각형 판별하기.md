### 내가 푼 풀이
```javascript
function solution(a, b, c) {
  let answer = 'YES';
  
  const arr = [a, b, c];
  arr.sort();
  if(a + b <= c) {
    answer = 'NO';
  }
  
  return answer;
}

console.log(solution(2, 5, 1));
```

### 해답
```javascript
function solution(a, b, c) {
  let answer = "YES", max;
  let tot = a + b + c;
  
  if(a>b) max = a;
  else max =. b;
  if(c > max). max = c;
  
  if(tot - max <= max) answer = "NO"; 
  
  return answer;
}

console.log(solution(2, 5, 1));
```
